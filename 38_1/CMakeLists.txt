cmake_minimum_required(VERSION 3.5)

project(38_1 VERSION 0.1 LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 COMPONENTS Widgets REQUIRED)
find_package(Qt${QT_VERSION_MAJOR} COMPONENTS Widgets REQUIRED)
find_package(Qt${QT_VERSION_MAJOR} COMPONENTS Multimedia MultimediaWidgets REQUIRED)

set(PROJECT_SOURCES
        main.cpp
        mainwindow.cpp
        mainwindow.h
)

set(PROJECT_RESOURCES
        res/up.jpg
        res/down.jpg
        res/paw.mp3
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(38_1
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )

else()
    if(ANDROID)
        add_library(38_1 SHARED
            ${PROJECT_SOURCES}
        )

    else()
        add_executable(38_1
            ${PROJECT_SOURCES} ${PROJECT_RESOURCES}
        )
    endif()
endif()

target_link_libraries(38_1 PRIVATE Qt${QT_VERSION_MAJOR}::Widgets)
target_link_libraries(38_1 PRIVATE Qt${QT_VERSION_MAJOR}::MultimediaWidgets
    Qt${QT_VERSION_MAJOR}::Multimedia)

set_target_properties(38_1 PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER my.example.com
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
)

if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(38_1)
endif()
